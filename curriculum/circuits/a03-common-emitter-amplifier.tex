\begin{circuitikz}[european resistors]
  \coordinate (B) at (5.0,4.2);
  \node[npn, anchor=B] (Q1) at (B) {};

  % Signal path on the base-height rail.
  \draw (0,0) node[ground] {}
    to[sV, l_={$v_{\mathrm{sig}}$}] (0,0 |- Q1.B)
    to[R, l={$R_{\mathrm{sig}}$}] (1.9,0 |- Q1.B)
    to[C, l={$C_{\mathrm{IN}}$}] (3.6,0 |- Q1.B)
    -- (Q1.B);
  \draw (0,0 |- Q1.B) -- (0,4.95)
    node[ocirc, label={[font=\small]above:{TPIN}}] {};

  % Divider and collector branches share a clean top supply rail.
  \coordinate (BDIV) at (4.25,0 |- Q1.B);
  \draw (BDIV) to[R, l_={$R_1$}] (4.25,7.0);
  \draw (BDIV) to[R, l={$R_2$}] (4.25,0) node[ground] {};
  \draw (BDIV) -- (Q1.B);
  \draw (Q1.C) to[R, l={$R_C$}] (Q1.C |- 0,7.0);
  \draw (4.25,7.0) -- (Q1.C |- 0,7.0)
    node[midway, vcc] {$V_{CC}$};

  % Split emitter resistance and bypass on exact terminal projections.
  \draw (Q1.E) to[R, l_={$R_{E1}$}] ++(0,-1.45)
    coordinate (EJ)
    to[R, l_={$R_{E2}$}] (Q1.E |- 0,0) node[ground] {};
  \draw (EJ) -- (8.0,0 |- EJ)
    to[C, l={$C_E$}] (8.0,0) node[ground] {};

  % Output rail at the exact collector height.
  \draw (Q1.C) -- (9.0,0 |- Q1.C)
    to[C, l={$C_{\mathrm{OUT}}$}] (10.9,0 |- Q1.C)
    -- (12.0,0 |- Q1.C)
    coordinate (OUT)
    to[R, l={$R_L$}] (12.0,0) node[ground] {};

  % Orthogonal test-point stubs.
  \draw (Q1.B) -- (5.0,4.95)
    node[ocirc, label={[font=\small]above:{TPB}}] {};
  \draw (8.0,0 |- Q1.C) -- (8.0,5.65)
    node[ocirc, label={[font=\small]above:{TPC}}] {};
  \draw (Q1.E) -- (7.35,0 |- Q1.E)
    node[ocirc, label={[font=\small]right:{TPE}}] {};
  \draw (OUT) -- (12.8,0 |- OUT)
    node[ocirc, label={[font=\small]right:{TPO}}] {};
  \draw (Q1.E |- 0,0) -- (7.2,0)
    node[ocirc, label={[font=\small]right:{TP0: $0$ V}}] {};

  \node[circ] at (BDIV) {};
  \node[circ] at (Q1.B) {};
  \node[circ] at (Q1.C) {};
  \node[circ] at (Q1.E) {};
  \node[circ] at (8.0,0 |- Q1.C) {};
  \node[circ] at (OUT) {};
  \node[circ] at (EJ) {};
  \node[circ] at (0,0 |- Q1.B) {};
  \node[font=\small] at (6.9,4.15) {$Q_1$};
